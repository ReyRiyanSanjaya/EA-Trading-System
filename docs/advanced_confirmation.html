bool ESD_ConfirmationStochasticFilter(bool is_buy)
{
    double k_main[], d_signal[];
    ArraySetAsSeries(k_main, true);
    ArraySetAsSeries(d_signal, true);
    
    int stoch_handle = iStochastic(_Symbol, PERIOD_CURRENT, 14, 3, 3, MODE_SMA, STO_LOWHIGH);
    CopyBuffer(stoch_handle, 0, 0, 3, k_main);
    CopyBuffer(stoch_handle, 1, 0, 3, d_signal);
    
    double k = k_main[0];
    
    if (is_buy)
        return (k < ESD_RSIOverbought);  // BUY only if NOT overbought
    else
        return (k > ESD_RSIOversold);    // SELL only if NOT oversold
}

bool ESD_ConfirmationRejectionCandle(MqlRates &candle, bool is_bullish)
{
    double body = MathAbs(candle.close - candle.open);
    double range = candle.high - candle.low;
    
    if (range == 0) return false;
    
    double body_ratio = body / range;
    double upper_wick = candle.high - MathMax(candle.close, candle.open);
    double lower_wick = MathMin(candle.close, candle.open) - candle.low;
    
    if (is_bullish)
    {
        double lower_ratio = lower_wick / range;
        return (lower_ratio > 0.6 && body_ratio < 0.3);
    }
    else
    {
        double upper_ratio = upper_wick / range;
        return (upper_ratio > 0.6 && body_ratio < 0.3);
    }
}
